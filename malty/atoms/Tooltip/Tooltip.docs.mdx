import { MaltyThemeProvider } from '@carlsberggroup/malty.theme.malty-theme-provider';
import { Tooltip, Position } from '.';

Text example:

A simple Tooltip component with some text

```js live
<MaltyThemeProvider theme="global">
  <Tooltip position={Position.Top} content={<span>Contents</span>} isOpen>
    <button type="button" onClick={() => null}>
      Toggle
    </button>
  </Tooltip>
</MaltyThemeProvider>
```
