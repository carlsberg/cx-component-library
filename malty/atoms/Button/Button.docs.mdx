import { IconNamesTypes } from '@carlsberggroup/malty.atoms.icon';
import { Button, ButtonType } from '.';

Button example:

A simple Button with `buttonType={Primary}`

```js live
<Button buttonType={ButtonType.Primary} text="Primary button" />
```

A simple Button with `buttonType={Secondary}`

```js live
<Button buttonType={ButtonType.Secondary} text="Secondary button" />
```

A simple Button with `buttonType={Floater}`

```js live
<Button buttonType={ButtonType.Floater} icon={IconNamesTypes.ArrowSmallUp} />
```

A simple Button with `buttonType={Link}` and text as a child

```js live
<Button buttonType={ButtonType.Link}>Link button</Button>
```
